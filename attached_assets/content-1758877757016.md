```
from perplexity import Perplexity

client = Perplexity()

search = client.search.create(
    query=[\
      "What is Comet Browser?",\
      "Perplexity AI",\
      "Perplexity Changelog"\
    ]
)

for result in search.results:
    print(f"{result.title}: {result.url}")
```
